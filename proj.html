import got3 as got
lst = list()

data = pd.read_csv('projectWithoutSemicolon.csv')

for i, name in enumerate(data.name):
    cnt = 0
    try:
        tweetCriteria = got.manager.TweetCriteria().setQuerySearch(name).setMaxTweets(10000)
        tweet = got.manager.TweetManager.getTweets(tweetCriteria)
        for v in tweet:
            if(name in v.text):
                cnt+=1
        data.at[i,'TweetCnt'] = cnt  
        lst.append(cnt)
    except:
        data.at[i,'TweetCnt'] = cnt
